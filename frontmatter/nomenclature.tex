\subsubsection*{Variables and functions}
\begingroup
\renewcommand{\arraystretch}{1.2}
\renewcommand{\tabularxcolumn}[1]{p{#1}}
\begin{longtable}{@{}p{2.5cm}p{\dimexpr\textwidth-2.5cm-2\tabcolsep}@{}}
    $Z(\omega)$ & Complex impedance as a function of angular frequency.\\
    $V(\omega)$ & Phasor voltage.\\
    $I(\omega)$ & Phasor current.\\
    $Z'(\omega)$ & Real component of impedance (resistive behaviour).\\
    $Z''(\omega)$ & Imaginary component of impedance (capacitive/inductive behaviour).\\
    $|Z|$ & Magnitude of impedance.\\
    $\phi$ & Phase angle of impedance.\\
    $\omega$ & Angular frequency (rad/s).\\
    $f$ & Frequency (Hz).\\
    $R_s$ & Solution resistance (series resistance).\\
    $R_{CT}$ & Charge transfer resistance.\\
    $C_{DL}$ & Double-layer capacitance.\\
    $Z_W$ & Warburg impedance.\\
    $Z_{CPE}$ & Constant phase element impedance.\\
    $T$ & CPE constant related to capacitance.\\
    $\alpha$ & CPE exponent characterising deviation from ideal capacitive behaviour ($0 \leq \alpha \leq 1$).\\
    $R_f$ & Feedback resistor.\\
    $R_{in}$ & Input resistor.\\
    $A_v$ & Voltage gain.\\
    $A_{noise}$ & Noise gain accounting for non-ideal feedback effects.\\
    $A_{OL}$ & Open loop gain.\\
    $A_{CL}$ & Closed loop gain.\\
    $\text{GBW}$ & Gain-bandwidth product.\\
    $f_c$ & Corner frequency / critical frequency.\\
    $\omega_0$ & Corner angular frequency.\\
    $\text{PM}$ & Phase margin.\\
    $\beta$ & Feedback factor.\\
    $V_{out}$ & Output voltage.\\
    $V_{in}$ & Input voltage.\\
    $V_{fb}$ & Feedback voltage.\\
    $V_n$ & Voltage at inverting input of op-amp.\\
    $V_p$ & Voltage at non-inverting input of op-amp.\\
    $I_n$ & Input bias current at inverting input.\\
    $I_p$ & Input bias current at non-inverting input.\\
    $I_{in}$ & Input current.\\
    $C_i$ & Combined input capacitance.\\
    $C_{in}$ & Op-amp input capacitance.\\
    $C_j$ & Junction capacitance.\\
    $X_C$ & Capacitive reactance.\\
    $\text{PSC}$ & Prescaler register value (16-bit).\\
    $\text{ARR}$ & Auto-reload register value (16-bit).\\
    $f_{target}$ & Target frequency for generation.\\
    $f_{meas}$ & Measured frequency.\\
    $f_{TIM}$ & Timer clock frequency.\\
    $f_{update}$ & Update frequency for DAC/ADC.\\
    $n$ & Number of bits of resolution or oversampling factor, depending on context.\\
\end{longtable}
\endgroup
